#running summary on africadata summary(africadata)

#summary of infant_mortality_and_life_expectancy summary(infant_mortality_and_life_expectancy) str(infant_mortality_and_life_expectancy)

#summary of population_and_life_expectancy summary(population_and_life_expectancy) str(population_and_life_expectancy)
